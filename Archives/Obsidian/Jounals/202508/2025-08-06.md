const page = dv.current();

// All tasks in the file
const allTasks = page.file.tasks;

// Filter for relevant tasks that match habit tags
const habitTasks = allTasks.filter(t =>
  habitTags.some(tag => t.text.includes(tag))
);

let expected = 0;
let completed = 0;

// Process each matching habit task
for (let task of habitTasks) {
  const tagMatch = habitTags.find(tag => task.text.includes(tag));
  if (!tagMatch) continue;

  const isWeekdayOnly = task.text.includes("#weekdays_only");
  const shouldCount = !isWeekdayOnly || isWeekday;

  if (shouldCount) {
    expected++;
    if (task.completed) completed++;
  }
}

// Percent and bar setup
const percent = expected > 0 ? Math.round((completed / expected) * 100) : 0;
const emoji = percent === 100 ? " 🎉" : "";
const barWidth = percent === 0 ? "4px" : `${percent}%`;

let barColor = "#f44336"; // red
if (completed === expected && expected > 0) barColor = "#30c750"; // green
else if (completed > 0) barColor = "#FFAA33"; // orange

// Render final callout
dv.paragraph(`
> [!progress] ${emoji}
> <div style="margin-top: 4px; margin-bottom: 4px; display: flex; align-items: center; gap: 8px;">
>   <div style="flex: 1; background: #e0e0e0; border-radius: 8px; height: 12px; overflow: hidden;">
>     <div style="width: ${barWidth}; background: ${barColor}; height: 100%;"></div>

const minBMI = 18.5;
const maxBMI = 24.9;
const safeWeeklyChange = 0.5; // kg per week

const gender = dv.current().gender;
const height = dv.current().height_in_cm;
const age = dv.current().age_in_years;
const activityLevel = dv.current().activity_level ?? 1;
const weightNum = dv.current().weight_in_kg ?? 0;

const activityFactors = { 1: 1.2, 2: 1.375, 3: 1.55, 4: 1.725, 5: 1.9 };
const heightMeters = height / 100;

// BMI and healthy weight range
const bmi = weightNum / (heightMeters ** 2);
const minIdealWeight = minBMI * (heightMeters ** 2);
const maxIdealWeight = maxBMI * (heightMeters ** 2);

// Weight difference relative to range
let weightDiff = 0;
let goalType = "maintain";

if (weightNum > maxIdealWeight) {
    weightDiff = weightNum - maxIdealWeight;
    goalType = "lose";
} else if (weightNum < minIdealWeight) {
    weightDiff = weightNum - minIdealWeight; // negative, need to gain
    goalType = "gain";
}

// Dynamic weight label
let weightLabel = "Weight Difference";
if (weightNum > maxIdealWeight) {
    weightLabel = "Extra Weight";
} else if (weightNum < minIdealWeight) {
    weightLabel = "Under Weight";
}

// Weeks to reach goal
const weeksToGoal = weightDiff !== 0 ? Math.abs(weightDiff) / safeWeeklyChange : 0;

// BMR and TDEE
let BMR = (10 * weightNum) + (6.25 * height) - (5 * age) + (gender === "male" ? 5 : -161);
let TDEE = BMR * (activityFactors[activityLevel] || 1.55);

// Daily calorie adjustment
let dailyChange = 0;
if (goalType === "lose") {
    const totalCaloriesToLose = weightDiff * 7700;
    dailyChange = totalCaloriesToLose / (weeksToGoal * 7);
} else if (goalType === "gain") {
    const totalCaloriesToGain = Math.abs(weightDiff) * 7700;
    dailyChange = totalCaloriesToGain / (weeksToGoal * 7);
}

const targetCalories = goalType === "lose" ? TDEE - dailyChange :
                       goalType === "gain" ? TDEE + dailyChange :
                       TDEE;

// Macronutrient distribution
let proteinPerKg = 0.75; // default
if (goalType === "gain" || goalType === "lose") {
    proteinPerKg = 0.5 + activityLevel * 0.25; // 0.75–1.75 g/kg roughly
}
const proteinGrams = Math.round(proteinPerKg * weightNum);

// Fat: 25–30% of calories
const fatCalories = targetCalories * 0.3;
const fatGrams = Math.round(fatCalories / 9);

// Carbs: remaining calories
const carbCalories = targetCalories - (proteinGrams * 4 + fatGrams * 9);
const carbGrams = Math.round(carbCalories / 4);

// Colors
const prefersDark = window.matchMedia('(prefers-color-scheme: dark)').matches;
const colorGreen = "#22c55e";
const colorRed = "#ef4444";
const colorBlue = "#3b82f6";
const colorPurple = "#8b5cf6";
const colorGoal = "#f97316"; // distinct orange for goal card

// BMI color
const getBMIcolor = (bmi) => bmi < minBMI || bmi > maxBMI ? colorRed : colorGreen;

// BMI category function
const getBMICategory = (bmi) => {
    if (bmi < 18.5) return "Underweight";
    if (bmi < 25) return "Healthy Weight";
    if (bmi < 30) return "Overweight";
    return "Obese";
};

// Stats array
const stats = [
  { label: "Body Mass Index", value: `${bmi.toFixed(1)} (${getBMICategory(bmi)})`, unit: "", color: getBMIcolor(bmi) },
  { label: "Healthy Weight Range", value: `${minIdealWeight.toFixed(0)}–${maxIdealWeight.toFixed(0)}`, unit: "kg", color: colorGreen },
  { label: weightLabel, value: Math.abs(weightDiff).toFixed(1), unit: "kg", color: weightDiff === 0 ? colorGreen : colorRed },
  { label: "Goal Timeline", value: weeksToGoal > 0 ? Math.ceil(weeksToGoal) : 0, unit: "weeks", color: colorBlue },
  { label: "BMR", value: BMR.toFixed(0), unit: "cal", color: colorGreen },
  { label: "TDEE", value: TDEE.toFixed(0), unit: "cal", color: colorGreen },
  { label: goalType === "lose" ? "Daily Calorie Deficit" : goalType === "gain" ? "Daily Calorie Surplus" : "Calorie Target", value: dailyChange.toFixed(0), unit: "cal", color: colorRed },
  { label: "Target Calories", value: targetCalories.toFixed(0), unit: "cal", color: colorBlue },
  { label: "Protein", value: proteinGrams.toFixed(0), unit: "g", color: colorPurple },
  { label: "Fat", value: fatGrams.toFixed(0), unit: "g", color: colorPurple },
  { label: "Carbs", value: carbGrams.toFixed(0), unit: "g", color: colorPurple },
  { label: "Goal", value: `${goalType === "lose" ? "Lose Weight" : goalType === "gain" ? "Gain Weight" : "Maintain Weight"}`, unit: "", color: colorGoal }
];

// Render cards
const container = dv.container;
container.innerHTML = "";
container.style.display = "grid";
container.style.gridTemplateColumns = "repeat(auto-fit, minmax(140px, 1fr))";
container.style.gap = "8px";
container.style.padding = "8px";
container.style.margin = "0 auto";

if (weightNum === 0) {
    const warning = document.createElement("div");
    warning.textContent = "⚠️ Enter today's weight to calculate targets.";
    warning.style.gridColumn = "1 / -1";
    warning.style.padding = "8px";
    warning.style.background = prefersDark ? "#4b5563" : "#fee2e2";
    warning.style.color = prefersDark ? "#f9fafb" : "#991b1b";
    warning.style.borderRadius = "8px";
    warning.style.fontSize = "0.9em";
    container.appendChild(warning);
} else {
    stats.forEach(stat => {
        const card = document.createElement("div");
        card.style.padding = "8px";
        card.style.borderRadius = "8px";
        card.style.background = "var(--background-primary)";
        card.style.boxShadow = prefersDark
            ? "0 1px 5px rgba(0,0,0,0.25)"
            : "0 1px 5px rgba(0,0,0,0.08)";
        card.style.display = "flex";
        card.style.flexDirection = "column";
        card.style.alignItems = "flex-start";
        card.style.borderLeft = `3px solid ${stat.color}`;

        value.textContent = `${stat.value} ${stat.unit}`;
        value.style.fontSize = "0.9em";
        value.style.fontWeight = "bold";
        value.style.color = stat.color;

        const label = document.createElement("div");
        label.textContent = stat.label;
        label.style.fontSize = "0.75em";
        label.style.color = "var(--text-muted)";

        card.appendChild(value);
        card.appendChild(label);
        container.appendChild(card);
    });
}
